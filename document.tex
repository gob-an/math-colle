	\begin{align*}
	\displaystyle \sum_{k=1}^{2n} \frac{(-1)^{k+1}} k
& = \sum_{\stackrel{k=1}{\text{impair}}}^{2n} \frac 1 k + \sum_{\stackrel{k=1}{\text{pair}}}^{2n} \frac {-1} k \\ 
& = \sum_{k=1}^{2n} \frac 1 k - 2 \sum_{\stackrel{k=1}{\text{pair}}}^{2n} \frac 1 k \\ 
& = \sum_{k=1}^{2n} \frac 1 k - 2 \sum_{p=1}^{n} \frac 1 {2p} \\ 
& = \sum_{k=1}^{2n} \frac 1 k - \sum_{k=1}^{n} \frac 1 {k}\\
	\end{align*}
